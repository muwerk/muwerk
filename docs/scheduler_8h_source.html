299  static bool mqttmatch(const String pubstr, const String substr) {
308  if (pubstr == substr)
309  return true;
310 
311  // Attiny compile: core needs to be extended, add c_str():
312  // In WString.h add: const char *c_str() const { return buffer; }
313  const char *pub = (const char *)pubstr.c_str();
314  const char *sub = (const char *)substr.c_str();
315 
316  int lp = strlen(pub);
317  int ls = strlen(sub);
318 
319  bool wPos = true; // sub wildcard is legal now
320  int ps = 0;
321  for (int pp = 0; pp < lp; pp++) {
322  if (pub[pp] == '+' || pub[pp] == '#') {
323  return false; // Illegal wildcards in pub
324  }
325  if (wPos) {
326  wPos = false;
327  if (sub[ps] == '#') {
328  if (ps == ls - 1) {
329  return true;
330  } else {
331  return false; // In sub, # must not be followed by
332  // anything else
333  }
334  }
335  if (sub[ps] == '+') {
336  while (pp < lp && pub[pp] != '/')
337  ++pp;
338  ++ps;
339  if (pp == lp) {
340  if (ps == ls) {
341  return true;
342  } else if (!strcmp(&sub[ps], "/#")) {
343  return true;
344  }
345  }
346  }
347  } else {
348  if (sub[ps] == '+' || sub[ps] == '#') {
349  return false; // Illegal wildcard-position
350  }
351  }
352  if (pub[pp] != sub[ps] && strcmp(&sub[ps], "/#")) {
353  return false;
354  }
355  if (pub[pp] == '/')
356  wPos = true;
357  if (pp == lp - 1) {
358  if (ps == ls - 1) {
359  return true;
360  }
361  if (!strcmp(&sub[ps + 1], "/#") || !strcmp(&sub[ps + 1], "#") ||
362  !strcmp(&sub[ps + 1], "+")) {
363  return true;
364  }
365  return false;
366  }
367  ++ps;
368  }
369  if (ps == ls) {
370  return true;
371  } else {
372  return false;
373  }
374  }
